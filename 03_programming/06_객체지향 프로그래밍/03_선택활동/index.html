  • attack(target) ← 오버라이딩

2단계: 코드 구현

Python RPG 캐릭터 클래스 
import random

class Character:
    def __init__(self, name, hp, attack):
        self.__name = name
        self.__hp = hp
        self.__max_hp = hp
        self._attack_power = attack
        self.level = 1

    def get_name(self): return self.__name
    def get_hp(self): return self.__hp
    def get_max_hp(self): return self.__max_hp
    def is_alive(self): return self.__hp > 0

    def take_damage(self, amount):
        self.__hp = max(0, self.__hp - amount)
        print(f"{self.__name}이(가) {amount} 데미지를 받았습니다! (HP: {self.__hp}/{self.__max_hp})")

    def heal(self, amount):
        self.__hp = min(self.__max_hp, self.__hp + amount)
        print(f"{self.__name}이(가) {amount} HP를 회복했습니다!")

    def attack(self, target):  # 오버라이딩 대상
        dmg = self._attack_power
        print(f"{self.__name}이(가) 공격합니다! ({dmg} 데미지)")
        target.take_damage(dmg)

class Warrior(Character):
    def __init__(self, name):
        super().__init__(name, hp=150, attack=25)
        self.shield = 10

    def attack(self, target):   # 오버라이딩!
        dmg = self._attack_power + random.randint(0, 10)
        print(f"⚔️ {self.get_name()}이(가) 검으로 강타합니다! ({dmg} 데미지)")
        target.take_damage(dmg)

class Mage(Character):
    def __init__(self, name):
        super().__init__(name, hp=90, attack=40)
        self.mana = 100

    def attack(self, target):   # 오버라이딩!
        if self.mana >= 20:
            self.mana -= 20
            dmg = self._attack_power + random.randint(0, 20)
            print(f"🔮 {self.get_name()}이(가) 마법을 시전합니다! ({dmg} 데미지)")
            target.take_damage(dmg)
        else:
            print("마나가 부족합니다!")

# 게임 실행
warrior = Warrior("홍길동")
mage = Mage("이순신")

warrior.attack(mage)    # 다형성 동작!
mage.attack(warrior)    # 다형성 동작!

활동 정리

1

캡슐화 확인

Character 클래스에서 __hp, __name은 외부에서 직접 수정할 수 없고, get_hp()와 같은 getter 메서드로만 접근합니다.

2

상속 확인

Warrior, Mage, Archer는 Character의 heal(), is_alive() 등을 직접 구현하지 않고 상속받아 사용합니다.

3

다형성 확인

같은 attack(target) 호출이지만, 전사는 검 공격, 마법사는 마법 공격, 궁수는 화살 공격으로 다르게 동작합니다.
