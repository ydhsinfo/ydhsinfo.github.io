if score >= 90:
    print("A 등급")
elif score >= 80:
    print("B 등급")
else:
    print("C 등급")

for i in range(5):      # 반복문
    print(i, end=" ")

##################################################
# 3. 함수
##################################################
def calc_avg(scores):
    return sum(scores) / len(scores)

avg = calc_avg([85, 90, 78])  # 84.33

##################################################
# 4. 클래스 (OOP)
##################################################
class Student:
    def __init__(self, name, score):
        self.__name = name    # 캡슐화
        self.__score = score

    def get_grade(self):
        return "A" if self.__score >= 90 else "B"

class GradStudent(Student):   # 상속
    def __init__(self, name, score, research):
        super().__init__(name, score)
        self.research = research

s = Student("홍길동", 95)
print(s.get_grade())  # A

자주 묻는 질문 (FAQ)

리스트는 순서가 있는 데이터 모음으로 인덱스(0, 1, 2...)로 접근합니다. [85, 90, 78]처럼 []를 사용합니다.
딕셔너리는 키(key)와 값(value)의 쌍으로 구성되며 {"name": "홍길동", "score": 95}처럼 {}를 사용합니다. 순서보다는 이름(키)으로 접근합니다.
함수는 독립적으로 정의되는 코드 블록입니다. def func(): ...
메서드는 클래스 내부에 정의된 함수로, 첫 번째 매개변수가 반드시 self입니다. 메서드는 객체를 통해 호출됩니다.
super()는 자식 클래스에서 부모 클래스의 메서드를 호출할 때 사용합니다. 주로 __init__ 메서드에서 super().__init__()로 부모의 초기화를 수행하는 데 사용됩니다. 이렇게 하면 부모의 속성을 자식 클래스에서도 그대로 활용할 수 있습니다.
SDGs와 관련된 데이터를 수집·분석하거나, 문제 상황을 시뮬레이션하거나, 해결책을 자동화하는 프로그램을 만들 수 있습니다. 예: 탄소 발자국 계산기(SDG 13), 식량 분배 최적화(SDG 2), 에너지 사용량 모니터링(SDG 7) 등.
